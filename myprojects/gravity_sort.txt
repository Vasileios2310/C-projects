ΑΛΓΟΡΙΘΜΟΣ Gravity_Sort
    ΣΤΑΘΕΡΕΣ:
        M := 100
        N := 10

    ΜΕΤΑΒΛΗΤΕΣ:
        A[0..N-1]          // πίνακας εισόδου
        B[0..N-1]          // ταξινομημένος πίνακας
        counts[0..M]       // πίνακας μετρήσεων
        i, j, k
        x
        sum
        threshold

    // Αρχικοποίηση counts
    ΓΙΑ j AΠΟ 0 ΜΕΧΡΙ M ΥΠΟΛΟΓΙΣΕ
        counts[j] = 0
    ΤΕΛΟΣ_ΓΙΑ

    // Δίαβάζει τον table_Α ( ύψος της στήλης για ο κάθε στοιχείο)
    ΓΙΑ i ΑΠΟ 0 ΜΕΧΡΙ N-1 ΥΠΟΛΟΓΙΣΕ
        ΕΠΑΝΑΛΑΒΕ
            ΤΥΠΩΣΕ "Δώσε το στοιχείο ", i+1, " (0..", M, ")"
            ΔΙΑΒΑΣΕ x
            ΑΝ x δεν είναι αριθμός ΤΟΤΕ
                ΤΥΠΩΣΕ "Λάθος είσοδος, ξαναδώσε"
                ΚΑΘΑΡΙΣΕ την υπόλοιπη γραμμή εισόδου
            ΑΛΛΙΩΣ_ΑΝ x < 0 Ή x > M ΤΟΤΕ
                ΤΥΠΩΣΕ "Εκτός ορίων, ξαναδώσε"
            ΑΛΛΙΩΣ
                A[i] = x
                ΕΞΟΔΟΣ από την επανάληψη
            ΤΕΛΟΣ_ΑΝ
        ΜΕΧΡΙΣ_ΟΤΟΥ έγκυρο x
    ΤΕΛΟΣ_ΓΙΑ

    // Δημιουργία ενδιάμεσου πίνακα table_counts (αριθμός χάντρων σε κάθε επίπεδο)
    ΓΙΑ i από 0 ΜΕΧΡΙ N-1 ΥΠΟΛΟΓΙΣΕ
        ΓΙΑ j από 0 ΜΕΧΡΙ A[i]-1 ΥΠΟΛΟΓΙΣΕ
            counts[j] = counts[j] + 1
        ΤΕΛΟΣ_ΓΙΑ
    ΤΕΛΟΣ_ΓΙΑ

    // Δημιουργία table_B (εδω οι χάντρες πέφτουν)
    ΓΙΑ k από 1 ΜΕΧΡΙ N ΥΠΟΛΟΓΙΣΕ
        threshold = N - k + 1
        sum = 0

        ΓΙΑ j από 0 ΜΕΧΡΙ M ΥΠΟΛΟΓΙΣΕ
            ΑΝ counts[j] ≥ threshold ΤΟΤΕ
                sum = sum + 1
            ΤΕΛΟΣ_ΑΝ
        ΤΕΛΟΣ_ΓΙΑ

        B[k-1] = sum
    ΤΕΛΟΣ_ΓΙΑ

    //Εκτύπωση στοιχείων table_B
    ΤΥΠΩΣΕ "B = ["
    ΓΙΑ i από 0 ΜΕΧΡΙ N-1 ΚΑΝΕ
        ΑΝ i > 0 ΤΟΤΕ
            ΤΥΠΩΣΕ ", "
        ΤΕΛΟΣ_ΑΝ
        ΤΥΠΩΣΕ B[i]
    ΤΕΛΟΣ_ΓΙΑ
    ΤΥΠΩΣΕ "]"
