Αλγόριθμος Έλεγχος_Ισοτητας_Κύβων

ΜΕΤΑΒΛΗΤΕΣ
INTEGER: N, i
REAL: Sum1 , Sum2 , SumTemp

ΑΡΧΗ
    ΕΠΑΝΕΛΑΒΕ       // Αμυντικός προγραμματισμος ως προς το Ν που θέτει ο χρήστης , μεσω φιλικού μηνύματος
        ΤΥΠΩΣΕ ("Δώσε αριθμό μεγαλύτερο ή ίσο του 100 και μικρότερο ή ίσο του 300")     
        ΔΙΑΑΒΑΣΕ Ν
    ΜΕΧΡΙΣ_ΟΤΟΥ Ν>= 100 ΚΑΙ Ν<=300

    Sum1 := 0               // Αρχικοποίηση μεταβλητής για αποθηλευση αθροισματος κύβων : 1^3 + 2^3 + 3^3 + ... +N^3
    SumTemp := 0            // Αρχικοποίηση μεταβλητής για αποθηλευση αθροισματος : 1 + 2 + 3 + ... + N

    ΓΙΑ i απο 1 ΜΕΧΡΙ Ν     //Βρίσκουμε πρώτα το Sum1 και μετά το SumTemp, διατρέχοντας το πλήθος αριθμών Ν
        Sum1 := Sum1 + i^3
        SumTemp := SumTemp + i 
    ΓΙΑ ΤΕΛΟΣ

    Sum2 := SumTemp ^2      // Βρίσκουμε το Sum2 , με τετραγωνισμό του SumTemp

    ΤΥΠΩΣΕ("Το άθροισμα Sum1 μέ βάση τον Τύπο 1 είναι:", Sum1)
    ΤΥΠΩΣΕ("Το άθροισμα Sum2 μέ βάση τον Τύπο 2 είναι:", Sum2)

    EAN Sum1 = Sum2 TOTE    // Κάνουμε έλεγχο θεωρήματος Sum1 και Sum2
        ΤΥΠΩΣΕ ("Το θεώρημα ισχύει")
    ΑΛΛΙΩΣ
         ΤΥΠΩΣΕ ("Το θεώρημα δεν ισχύει")
    ΕΑΝ ΤΕΛΟΣ
ΤΕΛΟΣ
